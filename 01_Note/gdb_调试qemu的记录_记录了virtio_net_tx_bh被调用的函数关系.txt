cfgqemu () 
{ 
    T=$(gettop);
    if [ ! "$T" ]; then
        echo "Couldn't locate the top of the tree.  Try setting TOP.";
        return;
    fi;
    QEMU_SRC=$T/vendor/sprd/qemu;
    QEMU_CROSS_COMPILER=$T/prebuilts/gcc/linux-x86/arm/gcc-linaro-arm-linux-gnueabihf-4.8;
    cd $QEMU_SRC;
    echo "Config QEMU";
    make distclean;
    ./configure --target-list=arm-softmmu --cross-prefix=$QEMU_CROSS_COMPILER/bin/arm-linux-gnueabihf- --enable-fdt --enable-kvm --disable-werror --extra-cflags="-march=armv7-a" --static;
    echo "Config QEMU Complete";
    cd $T
}
